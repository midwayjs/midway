@Provide('extend:GraphQLKoaMiddleware')

        // ç›´æ¥ä¼ å…¥ containerï¼Œæ„å‘³ç€ä½¿ç”¨åº”ç”¨ä¸Šä¸‹æ–‡ä½œä¸ºå®¹å™¨

        container: ({ context }: ResolverData<{ container: IMidwayContainer }>) => context.container,
